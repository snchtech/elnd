$(document).ready(function() {
  // Click event handler for the plus icon
  $('.plus-icon').on('click', function() {
    var clickedDiv = $(this).closest('.div-container'); // Get the clicked div
    var formRegion = $('#form-region').clone(); // Clone the form region

    // Insert the form region into the clicked div
    clickedDiv.append(formRegion);

    // Insert the form region into other divs
    $('.div-container').not(clickedDiv).each(function() {
      $(this).append(formRegion.clone());
    });
  });
});
