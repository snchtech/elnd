const inputElement = document.getElementById('yourInputId'); // Replace 'yourInputId' with the actual ID of your input element

inputElement.addEventListener('input', function() {
    const cyrillicRegex = /[\u0400-\u04FF]/; // Regular expression to match Cyrillic characters

    if (cyrillicRegex.test(inputElement.value)) {
        alert('Input contains Cyrillic characters.');
    } else {
        alert('Input does not contain Cyrillic characters.');
    }
});
