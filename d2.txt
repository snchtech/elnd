// Sample data
const menuItems = [
  { category: 'Personal', subcategory: 'Group1', name: 'Link name1' },
  { category: 'Personal', subcategory: 'Group1', name: 'Link name3' },
  { category: 'Personal', subcategory: 'Group2', name: 'Link name4' },
  { category: 'Position', subcategory: 'Group1', name: 'Link name5' },
  { category: 'Position', subcategory: 'Group1', name: 'Link name6' },
  { category: 'Position', subcategory: 'Group2', name: 'Link name7' }
];

// Group the items by category and subcategory
const groupedMenu = menuItems.reduce((acc, item) => {
  acc[item.category] = acc[item.category] || {};
  acc[item.category][item.subcategory] = acc[item.category][item.subcategory] || [];
  acc[item.category][item.subcategory].push(item.name);
  return acc;
}, {});

console.log(groupedMenu);
