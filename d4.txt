// Assuming you have the following tree structure:
const item = {
  Level1: {
    Level2: {
      MenuItem: {
        link: "your_link_value",
        label: "Menu Item Label"
      }
    }
  }
};

// Function to recursively build the HTML menu
function buildMenu(menuItem) {
  let html = '<ul>';

  for (const key in menuItem) {
    const item = menuItem[key];
    html += '<li>';

    // Check if the item has a link property
    if (item.link) {
      html += `<a href="${item.link}">${item.label}</a>`;
    } else {
      html += item.label;
    }

    // Check if the item has nested submenus
    if (typeof item === 'object') {
      html += buildMenu(item);
    }

    html += '</li>';
  }

  html += '</ul>';
  return html;
}

// Call the buildMenu function with the Level1 object
const menuHTML = buildMenu(item.Level1);

// Output the generated HTML
console.log(menuHTML);
