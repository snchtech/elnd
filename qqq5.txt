DECLARE
  l_sort_order NUMBER;
BEGIN
  FOR i IN (
    SELECT sqn, sort_order
    FROM apex_collection
    ORDER BY sqn
  ) LOOP
    l_sort_order := i.sort_order;
    UPDATE TABLE_1
    SET sort_order = l_sort_order
    WHERE id_tsk = i.id_tsk
    AND position = i.position
    AND department = i.department
    AND office = i.office
    AND stage = i.stage;
  END LOOP;
END;
