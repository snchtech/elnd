$(document).ready(function() {
  $('#firstSelect').change(function() {
    // Get the selected value of the first select item
    var selectedValue = $(this).val();

    // Check the selected value and set the "required" option for the second select item accordingly
    if (selectedValue === 'some_value') {
      $('#secondSelect').prop('required', true);
    } else {
      $('#secondSelect').prop('required', false);
    }
  });
});
