CREATE TABLE DOCUMENTS (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    entity_type VARCHAR2(50) CHECK (entity_type IN ('ALBUM', 'SCHEME')), -- Вказує, до чого належить документ
    entity_id NUMBER NOT NULL, -- Універсальний ідентифікатор (альбом або схема)
    file_name VARCHAR2(255),
    document_file BLOB,
    date_update DATE,
    editor_number NUMBER, -- Для схем, можна залишити NULL для альбомів
    document_type VARCHAR2(100),
    is_latest NUMBER(1) DEFAULT 0,
    CONSTRAINT fk_documents_album FOREIGN KEY (entity_id) REFERENCES ALBUM(id) ON DELETE CASCADE ENABLE,
    CONSTRAINT fk_documents_scheme FOREIGN KEY (entity_id) REFERENCES SCHEME(id) ON DELETE CASCADE ENABLE
);
