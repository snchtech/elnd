CREATE TABLE ALBUM_DOCUMENTS (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    album_id NUMBER NOT NULL,
    filename VARCHAR2(255) NOT NULL,
    mimetype VARCHAR2(100), -- Тип MIME для визначення формату файлу
    charset VARCHAR2(50), -- Кодування файлу (UTF-8, Windows-1251 тощо)
    file BLOB, -- Сам файл у форматі BLOB
    last_update DATE DEFAULT SYSDATE, -- Остання зміна файлу
    date_deleted DATE, -- Дата видалення (NULL = файл активний)
    uploaded_by NUMBER, -- ID користувача, який завантажив файл
    CONSTRAINT fk_album_documents_album FOREIGN KEY (album_id) REFERENCES ALBUM(id)
);
