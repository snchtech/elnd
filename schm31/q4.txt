CREATE TABLE SCHEME_NOTE (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    scheme_version_id NUMBER NOT NULL, -- Вказує, до якої версії схеми належить примітка
    note_text VARCHAR2(4000) NOT NULL, -- Текст примітки
    sort_order NUMBER, -- Порядок відображення примітки
    creator_id NUMBER NOT NULL, -- ID користувача, який створив примітку
    date_create DATE DEFAULT SYSDATE, -- Дата створення
    date_deleted DATE, -- Дата видалення (NULL - примітка активна)
    CONSTRAINT fk_scheme_note_scheme FOREIGN KEY (scheme_version_id) REFERENCES SCHEME(id)
);
