JavaScript:
javascript
const customScroll = document.querySelector('.custom-scroll');
customScroll.addEventListener('scroll', function() {
  const scrollThumb = customScroll.querySelector('.custom-scroll::-webkit-scrollbar-thumb');
  const scrollPercentage = (this.scrollTop / (this.scrollHeight - this.clientHeight)) * 100;
  scrollThumb.style.height = scrollPercentage + '%';
});
