```sql
  -- 1. –ö–æ—Ä–µ–Ω–µ–≤—ñ –∑–∞–ø–∏—Å–∏

    c.n001 AS task_id,

  UNION ALL

  -- 2. –†–µ–∫—É—Ä—Å–∏–≤–Ω–µ –ø—ñ–¥–∫–ª—é—á–µ–Ω–Ω—è –¥–æ—á—ñ—Ä–Ω—ñ—Ö –ø—ñ–¥–∑–∞–≤–¥–∞–Ω—å

    child.seq_id,
    child.n001,
    child.c001,
    child.c008,
    TO_NUMBER(child.c009),
    parent.level + 1,
    LPAD(' ', parent.level * 4, ' ') || child.c001
  FROM apex_collections child
  JOIN subtask_tree parent
    ON TO_NUMBER(child.c009) = parent.seq_id
  WHERE child.collection_name = 'TASK_SUBTASKS_COLLECTION'
)

  seq_id,
  task_id,
  content,
  subtask_order,
  parent_seq_id,
  level,
  display_content
FROM subtask_tree
ORDER BY subtask_order
```
