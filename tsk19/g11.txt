  const match = dateStr.match(/^(\d{2})\.(\d{2})\.(\d{4})[ ]+(\d{2}):(\d{2})$/);

}

function compareApexDates(startStr, endStr) {
  const start = parseApexDate(startStr);
  const end = parseApexDate(endStr);

  if (!start || !end) return null;

  if (start > end) return 1;
  if (start < end) return -1;
  return 0;
}
