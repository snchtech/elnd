  var item = apex.item(itemName);
  if (item && item.node.length > 0) {
    var el = item.node[0];

      el.setAttribute("required", "required");
      el.classList.add("is-required");

      el.removeAttribute("required");
      el.classList.remove("is-required");
    }
  }
