DECLARE
  l_deleted_per_id NUMBER;
BEGIN
  FOR i IN 1..APEX_APPLICATION.G_F01.COUNT LOOP
    -- Check if the PER_ID value exists in the Apex collection
    SELECT PER_ID
    INTO l_deleted_per_id
    FROM DUAL
    WHERE NOT EXISTS (
      SELECT 1
      FROM TABLE_2
      WHERE PER_ID = APEX_APPLICATION.G_F01(i)
    );

    -- If the PER_ID value does not exist in the Apex collection, delete it from the TABLE_2 table
    IF l_deleted_per_id IS NOT NULL THEN
      DELETE FROM TABLE_2
      WHERE PER_ID = l_deleted_per_id;
      DBMS_OUTPUT.PUT_LINE('Deleted PER_ID value: ' || l_deleted_per_id);
    END IF;
  END LOOP;
END;
