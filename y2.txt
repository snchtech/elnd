<input type="checkbox" id="accessCheckbox" onchange="filterMenu()">
<label for="accessCheckbox">Show Accessible Items Only</label>



function filterMenu() {
  var checkbox = document.getElementById('accessCheckbox');
  var liItems = document.querySelectorAll('li');

  liItems.forEach(function(li) {
    if (checkbox.checked && !li.classList.contains('access')) {
      li.style.display = 'none';
    } else {
      li.style.display = '';
    }
  });
}




function collapseMenu() {
  var ul = document.querySelector('ul');
  var nestedUl = ul.querySelectorAll('ul');
  
  nestedUl.forEach(function(item) {
    item.style.display = 'none';
  });
}
